-- ── UTILITIES ──────────────────────────────────────
local function countItem(name)
	local n = 0
	for _, t in ipairs(player.Backpack:GetChildren()) do
		if t.Name == name then n += 1 end
	end
	local ch = player.Character
	if ch then
		for _, t in ipairs(ch:GetChildren()) do
			if t:IsA("Tool") and t.Name == name then n += 1 end
		end
	end
	return n
end

local function countAllMS()
	return countItem(CFG.ITEM_MS_SMALL) + countItem(CFG.ITEM_MS_MEDIUM) + countItem(CFG.ITEM_MS_LARGE)
end

local function getEquippableMS()
	if countItem(CFG.ITEM_MS_SMALL)  > 0 then return CFG.ITEM_MS_SMALL  end
	if countItem(CFG.ITEM_MS_MEDIUM) > 0 then return CFG.ITEM_MS_MEDIUM end
	if countItem(CFG.ITEM_MS_LARGE)  > 0 then return CFG.ITEM_MS_LARGE  end
	return nil
end

local function hasAllIngredients()
	return countItem(CFG.ITEM_WATER) >= 1
		and countItem(CFG.ITEM_SUGAR) >= 1
		and countItem(CFG.ITEM_GEL)   >= 1
end

local function equipTool(name)
	local ch = player.Character
	if not ch then return false end
	local hum = ch:FindFirstChildOfClass("Humanoid")
	local t   = player.Backpack:FindFirstChild(name)
	if hum and t then hum:EquipTool(t) task.wait(0.2) return true end
	return false
end

local function unequipAll()
	local ch = player.Character
	if not ch then return end
	local hum = ch:FindFirstChildOfClass("Humanoid")
	if hum then hum:UnequipTools() end
end

local function pressE()
	pcall(function()
		VIM:SendKeyEvent(true,  Enum.KeyCode.E, false, game)
		task.wait(0.15)
		VIM:SendKeyEvent(false, Enum.KeyCode.E, false, game)
	end)
end

local function safeFireProximity(obj)
	pcall(function()
		if fireproximityprompt then
			fireproximityprompt(obj)
		else
			local event = obj.PromptButtonHoldBegan
			if event then
				VIM:SendKeyEvent(true, obj.KeyCode, false, game)
				task.wait(0.1)
				VIM:SendKeyEvent(false, obj.KeyCode, false, game)
			end
		end
	end)
end

local function firePromptNearby(radius)
	local ch   = player.Character
	local root = ch and ch:FindFirstChild("HumanoidRootPart")
	if not root then return end
	radius = radius or 8
	for _, obj in ipairs(workspace:GetDescendants()) do
		if obj:IsA("ProximityPrompt") then
			local part = obj.Parent
			if part and part:IsA("BasePart") then
				if (root.Position - part.Position).Magnitude <= radius then
					safeFireProximity(obj)
				end
			end
		end
	end
end

local function fireCookingPotNearby(radius)
	local ch   = player.Character
	local root = ch and ch:FindFirstChild("HumanoidRootPart")
	if not root then return end
	radius = radius or 15
	for _, obj in ipairs(workspace:GetDescendants()) do
		if obj:IsA("ProximityPrompt") then
			local anc      = obj.Parent
			local cookPart = nil
			while anc do
				if anc:IsA("BasePart") and anc.Name == "Cooking Pot" then
					cookPart = anc
					break
				end
				anc = anc.Parent
			end
			if cookPart and (root.Position - cookPart.Position).Magnitude <= radius then
				safeFireProximity(obj)
			end
		end
	end
end

-- ── AUTO JUAL ──────────────────────────────────────
local function doAutoSell(setStatus2)
	local msTotal = countAllMS()
	if msTotal == 0 then
		setStatus2("ℹ️ Tidak ada MS", Color3.fromRGB(160, 160, 180))
		return
	end
	setStatus2("💰 Memulai jual " .. msTotal .. " MS...", Color3.fromRGB(50, 210, 110))
	task.wait(0.3)
	local sold       = 0
	local maxFail    = 5
	local failStreak = 0
	while countAllMS() > 0 do
		local msName = getEquippableMS()
		if not msName then break end
		local ok = equipTool(msName)
		if not ok then
			failStreak += 1
			setStatus2("❌ Gagal equip (" .. failStreak .. "/" .. maxFail .. ")", Color3.fromRGB(210, 40, 40))
			task.wait(1)
			if failStreak >= maxFail then break end
			continue
		end
		local bS = countItem(CFG.ITEM_MS_SMALL)
		local bM = countItem(CFG.ITEM_MS_MEDIUM)
		local bL = countItem(CFG.ITEM_MS_LARGE)
		task.wait(0.2)
		pressE()
		task.wait(0.3)
		firePromptNearby(CFG.SELL_RADIUS)
		task.wait(0.3)
		pressE()
		local elapsed = 0
		local terjual = false
		while elapsed < CFG.SELL_TIMEOUT do
			local diff = (bS - countItem(CFG.ITEM_MS_SMALL))
				+ (bM - countItem(CFG.ITEM_MS_MEDIUM))
				+ (bL - countItem(CFG.ITEM_MS_LARGE))
			if diff > 0 then
				sold      += diff
				totalSold += diff
				terjual    = true
				failStreak = 0
				break
			end
			task.wait(0.3)
			elapsed += 0.3
		end
		if terjual then
			setStatus2("💰 Terjual " .. sold .. " | Sisa: " .. countAllMS(), Color3.fromRGB(50, 210, 110))
			task.wait(0.2)
		else
			failStreak += 1
			setStatus2("⚠️ Tidak terjual (" .. failStreak .. "/" .. maxFail .. ")", Color3.fromRGB(255, 155, 35))
			task.wait(1.2)
			if failStreak >= maxFail then
				setStatus2("❌ Gagal. Dekati NPC!", Color3.fromRGB(210, 40, 40))
				break
			end
		end
	end
	unequipAll()
	if sold > 0 then
		setStatus2("✅ Terjual " .. sold .. " MS (total: " .. totalSold .. ")", Color3.fromRGB(50, 210, 110))
	else
		setStatus2("⚠️ Tidak ada MS terjual. Dekati NPC!", Color3.fromRGB(255, 155, 35))
	end
	task.wait(1)
end

local buyQty  = {1, 1, 1}
local buyBusy = false

local BUY_ITEMS = {
	{name="Gelatin",         display="🟡 Gelatin"},
	{name="Sugar Block Bag", display="🧂 Sugar Block Bag"},
	{name="Water",           display="💧 Water"},

local function doAutoBuy(setStatus2, overrideQty)
	if not storePurchaseRE then
		setStatus2("❌ Remote tidak ada!", Color3.fromRGB(210, 40, 40))
		task.wait(1.5)
		return
	end
	local totalBought = 0
	for idx, item in ipairs(BUY_ITEMS) do
		local qty = overrideQty or buyQty[idx] or 1
		setStatus2("🛒 Beli " .. item.display .. " ×" .. qty .. "...", Color3.fromRGB(100, 180, 255))
		local before     = countItem(item.name)
		local successBuy = 0
		for i = 1, qty do
			local ok = pcall(function()
				storePurchaseRE:FireServer(item.name, 1)
			end)
			if ok then successBuy = successBuy + 1 end
			task.wait(0.4)
		end
		local timeout = 0
		local gained  = 0
		repeat
			task.wait(0.2)
			timeout = timeout + 0.2
			gained  = countItem(item.name) - before
		until gained >= qty or timeout > 6
		if gained < qty then
			local missing = qty - gained
			setStatus2("🔁 Retry " .. missing .. " " .. item.display, Color3.fromRGB(255, 160, 40))
			for i = 1, missing do
				pcall(function() storePurchaseRE:FireServer(item.name, 1) end)
				task.wait(0.5)
			end
			timeout = 0
			repeat
				task.wait(0.2)
				timeout = timeout + 0.2
				gained  = countItem(item.name) - before
			until gained >= qty or timeout > 5
		end
		totalBought = totalBought + gained
		totalBuy    = totalBuy    + gained
		if gained < qty then
			setStatus2("⚠️ " .. item.display .. " kurang (" .. gained .. "/" .. qty .. ")", Color3.fromRGB(255, 120, 120))
		else
			setStatus2("✅ " .. item.display .. " ×" .. gained .. " selesai!", Color3.fromRGB(80, 220, 130))
		end
		task.wait(0.2)
	end
	setStatus2("✅ Beli selesai! " .. totalBought .. " item.", Color3.fromRGB(80, 220, 130))
	task.wait(1)
end

-- ── AUTO MASAK ─────────────────────────────────────
local function countdown(secs, fmt, color)
	for i = secs, 1, -1 do
		if not isRunning then return false end
		setStatus(string.format(fmt, i), color)
		task.wait(1)
	end
	return true
end

local function cookInteract(toolName, radius)
	if toolName then
		equipTool(toolName)
		task.wait(0.2)
	end
	fireCookingPotNearby(radius or 8)
end

local rpcQueue = {}

if rpcRE then
	rpcRE.OnClientEvent:Connect(function(_, tblArg)
		if type(tblArg) ~= "table" then return end
		local v1  = tblArg[1]
		local v2  = tblArg[2]
		local msg = tostring(v1 or ""):lower()
		if v2 == "TextLabel" and tonumber(v1) then
			table.insert(rpcQueue, {type="timer", secs=tonumber(v1)})
			return
		end
		if msg:find("boil") or msg:find("water") then
			table.insert(rpcQueue, {type="wait_boil"})
		elseif msg:find("sugar") then
			table.insert(rpcQueue, {type="add_sugar"})
		elseif msg:find("gelatin") then
			table.insert(rpcQueue, {type="add_gelatin"})
		elseif msg:find("cook") then
			table.insert(rpcQueue, {type="wait_cook"})
		elseif msg:find("bag") then
			table.insert(rpcQueue, {type="bag_result"})
		end
	end)
end

local function waitRPC(instrType, timeout)
	local start = tick()
	while tick() - start < timeout do
		for i = 1, #rpcQueue do
			local inst = rpcQueue[i]
			if inst and inst.type == instrType then
				table.remove(rpcQueue, i)
				return inst
			end
		end
		task.wait(0.1)
	end
	return nil
end

local function popTimer()
	for i = 1, #rpcQueue do
		local v = rpcQueue[i]
		if v.type == "timer" then
			table.remove(rpcQueue, i)
			return v.secs
		end
	end
	return nil
end

local function doOneCook()
	isBusy = true
	table.clear(rpcQueue)
	local snapS = countItem(CFG.ITEM_MS_SMALL)
	local snapM = countItem(CFG.ITEM_MS_MEDIUM)
	local snapL = countItem(CFG.ITEM_MS_LARGE)
	setStatus("💧 Masukkan Water...", Color3.fromRGB(100, 180, 255))
	cookInteract(CFG.ITEM_WATER)
	local boilSecs
	for _ = 1, 30 do
		boilSecs = popTimer()
		if boilSecs then break end
		task.wait(0.1)
	end
	boilSecs = boilSecs or CFG.WATER_WAIT
	if not countdown(boilSecs, "💧 Mendidih... ⏱ %ds", Color3.fromRGB(80, 150, 255)) then
		isBusy = false
		return false
	end
	setStatus("🧂 Tunggu Sugar...", Color3.fromRGB(255, 220, 100))
	waitRPC("add_sugar", 10)
	setStatus("🧂 Masukkan Sugar...", Color3.fromRGB(255, 220, 100))
	cookInteract(CFG.ITEM_SUGAR)
	setStatus("🟡 Tunggu Gelatin...", Color3.fromRGB(255, 200, 50))
	waitRPC("add_gelatin", 10)
	setStatus("🟡 Masukkan Gelatin...", Color3.fromRGB(255, 200, 50))
	cookInteract(CFG.ITEM_GEL)
	local cookSecs
	for _ = 1, 30 do
		cookSecs = popTimer()
		if cookSecs then break end
		task.wait(0.1)
	end
	cookSecs = cookSecs or CFG.COOK_WAIT
	if not countdown(cookSecs, "🔥 Memasak... ⏱ %ds", Color3.fromRGB(80, 140, 255)) then
		isBusy = false
		return false
	end
	setStatus("🎒 Tunggu Bag...", Color3.fromRGB(100, 160, 255))
	waitRPC("bag_result", 12)
	local bag
	local t = 0
	repeat
		bag = player.Backpack:FindFirstChild(CFG.ITEM_EMPTY)
		task.wait(0.3)
		t += 0.3
	until bag or t > 10
	if not bag then
		setStatus("❌ Empty Bag tidak ada!", Color3.fromRGB(210, 40, 40))
		isBusy = false
		return false
	end
	setStatus("🎒 Ambil Marshmallow...", Color3.fromRGB(100, 180, 255))
	cookInteract(CFG.ITEM_EMPTY)
	local waitMS = 0
	local newS, newM, newL
	repeat
		task.wait(0.3)
		waitMS += 0.3
		newS = countItem(CFG.ITEM_MS_SMALL)  - snapS
		newM = countItem(CFG.ITEM_MS_MEDIUM) - snapM
		newL = countItem(CFG.ITEM_MS_LARGE)  - snapL
	until (newS > 0 or newM > 0 or newL > 0) or waitMS > 8
	if newS > 0 then
		stats.small  += newS
	elseif newM > 0 then
		stats.medium += newM
	elseif newL > 0 then
		stats.large  += newL
	else
		stats.small  += 1
	end
	setStatus("✅ MS ke-" .. totalMS() .. " selesai!", Color3.fromRGB(80, 210, 255))
	isBusy = false
	return true
end

	while isRunning do
		if not hasAllIngredients() then
			setStatus("❌ Bahan habis!", Color3.fromRGB(210, 40, 40))
			isRunning = false
			break
		end
		doOneCook()
		task.wait(0.3)
	end
end

local isTeleporting     = false

local function moveVehicle(vehicle, targetPos)
	local anchor = vehicle.PrimaryPart
		or vehicle:FindFirstChildOfClass("VehicleSeat")
		or vehicle:FindFirstChildOfClass("BasePart")
	if not anchor then return end
	local spawnPos = targetPos + Vector3.new(0, 0.5, 0)
	local newCF    = CFrame.new(spawnPos, spawnPos + Vector3.new(0, 0, 1))
	for _, p in ipairs(vehicle:GetDescendants()) do
		if p:IsA("BasePart") then
			pcall(function()
				p.AssemblyLinearVelocity  = Vector3.zero
				p.AssemblyAngularVelocity = Vector3.zero
				p.Anchored = true
			end)
		end
	end
	task.wait(0.05)
	if vehicle.PrimaryPart then
		vehicle:SetPrimaryPartCFrame(newCF)
	else
		anchor.CFrame = newCF
	end
	task.wait(0.05)
	for _, p in ipairs(vehicle:GetDescendants()) do
		if p:IsA("BasePart") then
			pcall(function()
				p.Anchored = false
				p.AssemblyLinearVelocity  = Vector3.zero
				p.AssemblyAngularVelocity = Vector3.zero
			end)
		end
	end
end

local function stepTeleport(targetPos)
	if isTeleporting then return end
	local ch  = player.Character
	local hum = ch and ch:FindFirstChildOfClass("Humanoid")
	if not ch or not hum then return end
	isTeleporting = true
	task.spawn(function()
		local seatPart = hum.SeatPart
		if seatPart then
			local vehicle = seatPart:FindFirstAncestorOfClass("Model")
			if vehicle then moveVehicle(vehicle, targetPos) end
		else
			local root = ch:FindFirstChild("HumanoidRootPart")
			if root then
				pcall(function() root.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 0)) end)
			end
		end
		isTeleporting = false
	end)
end

local function respawnTeleport(targetPos)
	if isTeleporting then return end
	isTeleporting     = true
	pendingRespawnPos = targetPos
	setStatus("☠️ Respawn ke titik...", Color3.fromRGB(255, 160, 40))
	local ch  = player.Character
	local hum = ch and ch:FindFirstChildOfClass("Humanoid")
	if hum then
		pcall(function() hum.Health = 0 end)
	else
		isTeleporting = false
	end
end

local function fullyTeleport(targetPos)
	local ch  = player.Character
	local hum = ch and ch:FindFirstChildOfClass("Humanoid")
	if not ch or not hum then task.wait(1) return end
	local seatPart = hum.SeatPart
	if seatPart then
		local vehicle = seatPart:FindFirstAncestorOfClass("Model")
		if vehicle then
			moveVehicle(vehicle, targetPos)
			task.wait(0.5)
		end
	else
		local root = ch:FindFirstChild("HumanoidRootPart")
		if root then
			pcall(function() root.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 0)) end)
		end
		task.wait(0.5)
	end
end

local originalBaseplate     = nil
local originalBaseplateSize = nil
local undergroundPart       = nil
local isUnderground         = false
local SURFACE_Y             = 3

local function findBaseplate()
	for _, v in ipairs(workspace:GetChildren()) do
		if v:IsA("Part") and v.Name:lower():find("baseplate") then
			return v
		end
	end
	return workspace:FindFirstChild("Baseplate")
end

local function resizeBaseplateForUnderground()
	local bp = findBaseplate()
	if not bp then return false end
	if not originalBaseplate then
		originalBaseplate     = bp
		originalBaseplateSize = bp.Size
	end
	pcall(function()
		bp.Size         = Vector3.new(2048, 5, 2048)
		bp.CFrame       = CFrame.new(0, -7.5, 0)
		bp.Anchored     = true
		bp.CanCollide   = true
		bp.Transparency = 0.5
		bp.Material     = Enum.Material.Granite
	end)
	return true
end

local function restoreBaseplate()
	if originalBaseplate and originalBaseplateSize then
		pcall(function()
			originalBaseplate.Size   = originalBaseplateSize
			originalBaseplate.CFrame = CFrame.new(0, -2, 0)
		end)
	end
end

local function teleportCharToPos(pos, underground)
	local ch   = player.Character
	local root = ch and ch:FindFirstChild("HumanoidRootPart")
	if not root then return false end
	local oldCollide = {}
	for _, part in ipairs(ch:GetDescendants()) do
		if part:IsA("BasePart") then
			oldCollide[part] = part.CanCollide
			part.CanCollide  = false
		end
	end
	pcall(function()
		root.CFrame = CFrame.new(pos + Vector3.new(0, underground and 2 or 2, 0))
	end)
	task.wait(0.1)
	for part, col in pairs(oldCollide) do
		if part and part.Parent then
			part.CanCollide = col

local function isPlayerNearby(radius)
	local ch   = player.Character
	local root = ch and ch:FindFirstChild("HumanoidRootPart")
	if not root then return false, nil end
	for _, plr in ipairs(Players:GetPlayers()) do
		if plr == player then continue end
		local oc    = plr.Character
		local oRoot = oc and oc:FindFirstChild("HumanoidRootPart")
		if oRoot then
			if (root.Position - oRoot.Position).Magnitude <= radius then
				return true, plr.Name
			end
		end
	end
	return false, nil
end

local altDeleteEnabled = false

local undoCountLabel   = nil

local function updateDeleteLabels()
	if deleteCountLabel then deleteCountLabel.Text = "Deleted: " .. deleteCount end
	if undoCountLabel   then undoCountLabel.Text   = "Undo: "    .. #deleteHistory end
end

local function pushDeleteUndo(part)
	if not part or not part:IsA("BasePart") then return end
	local info = {
		Name         = part.Name,
		ClassName    = part.ClassName,
		Size         = part.Size,
		CFrame       = part.CFrame,
		Color        = part.Color,
		Material     = part.Material,
		Transparency = part.Transparency,
		Anchored     = part.Anchored,
		CanCollide   = part.CanCollide,
		Parent       = part.Parent,
	}
	table.insert(deleteHistory, info)
	if #deleteHistory > MAX_UNDO then table.remove(deleteHistory, 1) end
	deleteCount += 1
	updateDeleteLabels()
end

local function undoLastDelete()
	if #deleteHistory == 0 then
		setStatus("⚠️ Tidak ada part untuk di-undo!", Color3.fromRGB(255, 200, 60))
		return false
	end
	local info = table.remove(deleteHistory)
	local ok, err = pcall(function()
		local newPart          = Instance.new(info.ClassName)
		newPart.Name           = info.Name
		newPart.Size           = info.Size
		newPart.CFrame         = info.CFrame
		newPart.Color          = info.Color
		newPart.Material       = info.Material
		newPart.Transparency   = info.Transparency
		newPart.Anchored       = info.Anchored
		newPart.CanCollide     = info.CanCollide
		newPart.Parent         = info.Parent or workspace
	end)
	if ok then
		setStatus("↩️ Undo: " .. info.Name .. " dikembalikan!", Color3.fromRGB(80, 210, 160))
	else
		setStatus("❌ Undo gagal: " .. tostring(err), Color3.fromRGB(210, 40, 40))
	end
	updateDeleteLabels()
	return ok
end

local function tryDeleteTarget(target)
	if not altDeleteEnabled then return end
	if target and target:IsA("BasePart") then
		local partName = target.Name
		pushDeleteUndo(target)
		pcall(function() target:Destroy() end)
		setStatus("🗑️ Deleted: " .. partName, Color3.fromRGB(255, 80, 80))
	end
end

local hitboxEnabled       = false
local hitboxSize          = Vector3.new(10, 10, 10)
local hitboxTransparency  = 1
local HB_TARGET_PARTS     = {"Torso", "UpperTorso", "LowerTorso"}
local hbOriginalProps     = {}
local hbTrackedPlayers    = {}
local hbNoCollConstraints = {}
local hbLoopActive        = true

local function hbGetLocalParts()
	local parts = {}
	local ch = player.Character
	if not ch then return parts end
	for _, v in ipairs(ch:GetDescendants()) do
		if v:IsA("BasePart") then table.insert(parts, v) end
	end
	return parts
end

local function hbApplyNoCollision(targetPart)
	if hbNoCollConstraints[targetPart] then return end
	local localParts = hbGetLocalParts()
	if #localParts == 0 then return end
	hbNoCollConstraints[targetPart] = {}
	for _, lp in ipairs(localParts) do
		pcall(function()
			local nc  = Instance.new("NoCollisionConstraint")
			nc.Part0  = targetPart
			nc.Part1  = lp
			nc.Parent = workspace
			table.insert(hbNoCollConstraints[targetPart], nc)
		end)
	end
end

local function hbRemoveAllNoCollision()
	for _, constraints in pairs(hbNoCollConstraints) do
		for _, nc in ipairs(constraints) do
			pcall(function() nc:Destroy() end)
		end
	end
	hbNoCollConstraints = {}
end

player.CharacterAdded:Connect(function(ch)
	task.wait(1)
	hbRemoveAllNoCollision()
	if hitboxEnabled then
		for _, plr in pairs(hbTrackedPlayers) do
			if plr and plr.Character then
				for _, pName in ipairs(HB_TARGET_PARTS) do
					local part = plr.Character:FindFirstChild(pName)
					if part and part:IsA("BasePart") then
						hbApplyNoCollision(part)
					end
				end
			end
		end
	end
end)

local function hbSaveProps(plr, part)
	if not hbOriginalProps[plr.Name] then hbOriginalProps[plr.Name] = {} end
	if not hbOriginalProps[plr.Name][part.Name] then
		hbOriginalProps[plr.Name][part.Name] = {
			Size         = part.Size,
			Transparency = part.Transparency,
			CanCollide   = part.CanCollide,
			Massless     = part.Massless,
		}
	end
end

local function hbRevertAll()
	for playerName, plr in pairs(hbTrackedPlayers) do
		if hbOriginalProps[playerName] and plr and plr.Parent then
			local ch = plr.Character
			if ch then
				for partName, props in pairs(hbOriginalProps[playerName]) do
					local part = ch:FindFirstChild(partName)
					if part and part:IsA("BasePart") then
						part.Size         = props.Size
						part.Transparency = props.Transparency
						part.CanCollide   = props.CanCollide
						part.Massless     = props.Massless
					end
				end
			end
		end
	end
	hbRemoveAllNoCollision()
end

for _, plr in ipairs(Players:GetPlayers()) do
	if plr ~= player then hbTrackedPlayers[plr.Name] = plr end
end

	hbTrackedPlayers[plr.Name] = plr

	hbOriginalProps[plr.Name]  = nil

						end
					end

	end
	fovContainer.Position = UDim2.new(0, cx, 0, cy)
	fovContainer.Size     = UDim2.new(0, dia, 0, dia)
	local segColor = (aimLocked and aimLockedPart)
		and Color3.fromRGB(255, 60, 60)
		or  Color3.fromRGB(255, 200, 60)
	local step = (2 * math.pi) / SEGS
	for i, seg in ipairs(fovSegments) do
		seg.BackgroundColor3 = segColor
		local angle     = (i - 1) * step
		local nextAngle = angle + step
		local ax = math.cos(angle)
		local ay = math.sin(angle)
		local bx = math.cos(nextAngle)
		local by = math.sin(nextAngle)
		local midX  = (ax + bx) / 2 * r
		local midY  = (ay + by) / 2 * r
		local dx    = (bx - ax) * r
		local dy    = (by - ay) * r
		local segL  = math.sqrt(dx * dx + dy * dy) + 0.5
		local rot   = math.deg(math.atan2(dy, dx))
		seg.Size     = UDim2.new(0, segL, 0, SEG_THICK)
		seg.Position = UDim2.new(0.5, midX - segL / 2, 0.5, midY - SEG_THICK / 2)
		seg.Rotation = rot
	end
end

local function getClosestPlayerOnScreen()
	local vp           = _camera.ViewportSize
	local screenCenter = Vector2.new(vp.X / 2, vp.Y / 2)
	local aimCenter
	if isMobileDevice() then
		aimCenter = screenCenter
	else
		local mousePos = UIS:GetMouseLocation()
		aimCenter = Vector2.new(mousePos.X, mousePos.Y)
	end
	local closestDist  = aimAssistRadius
	local closestPart  = nil
	for _, plr in ipairs(Players:GetPlayers()) do
		if plr == player then continue end
		local ch        = plr.Character
		if not ch then continue end
		local hrpTarget = ch:FindFirstChild("HumanoidRootPart")
		local hum       = ch:FindFirstChildOfClass("Humanoid")
		if not hrpTarget then continue end
		if hum and hum.Health <= 0 then continue end
		local screenPos, onScreen = _camera:WorldToViewportPoint(hrpTarget.Position)
		if onScreen then
			local dist = (Vector2.new(screenPos.X, screenPos.Y) - aimCenter).Magnitude
			if dist < closestDist then
				closestDist = dist
				closestPart = hrpTarget
			end
		end
	end
	return closestPart
end

UIS.InputBegan:Connect(function(input, gp)
	if gp then return end
	if isMobileDevice() then return end
	if not aimAssistEnabled then return end
	if input.UserInputType == Enum.UserInputType.MouseButton2 then
		if aimLocked then
			aimLocked     = false
			aimLockedPart = nil
			setStatus("🎯 Aim unlock", Color3.fromRGB(255, 200, 60))
		else
			local target = getClosestPlayerOnScreen()

				aimLocked     = true
				aimLockedPart = target

				local name = targetPlr and targetPlr.Name or "Target"
				setStatus("🔴 Aim LOCKED → " .. name, Color3.fromRGB(255, 60, 60))
			else
				setStatus("⚠️ Tidak ada target di FOV!", Color3.fromRGB(255, 155, 35))
			end
		end
	end
end)

-- ─── HELPER: Sub-tab builder (reusable) ───────────────────
local function buildSubTabs(parentFrame, tabNames, tabColors)
	-- returns: pages table, switchTab function
	local tabBar = mkFrame(parentFrame, C.tabBg, 3)
	tabBar.Size = UDim2.new(1, 0, 0, 30)
	local tabLine = mkFrame(parentFrame, C.line, 3)
	tabLine.Size     = UDim2.new(1, 0, 0, 1)
	tabLine.Position = UDim2.new(0, 0, 0, 30)
	tabLine.BackgroundTransparency = 0.5
	local tabBtns = {}
	local pages   = {}
	local tw      = 1 / #tabNames
	for i, name in ipairs(tabNames) do
		local col = tabColors and tabColors[i] or C.blue
		local tb = mkBtn(tabBar, name, C.txtD, Enum.Font.GothamBold, 4, 9)
		tb.BackgroundTransparency = 0
		tb.BackgroundColor3       = C.tabBg
		tb.Size     = UDim2.new(tw, 0, 1, 0)
		tb.Position = UDim2.new(tw * (i-1), 0, 0, 0)
		tabBtns[i]  = tb
		local ul = mkFrame(tb, col, 5)
		ul.Name     = "UL"
		ul.Size     = UDim2.new(0.6, 0, 0, 2)
		ul.Position = UDim2.new(0.2, 0, 1, -2)
		ul.Visible  = (i==1)
		corner(ul, 1)
		local pg = mkFrame(parentFrame, Color3.fromRGB(0,0,0), 2)
		pg.BackgroundTransparency = 1
		pg.Size             = UDim2.new(1, 0, 1, -31)
		pg.Position         = UDim2.new(0, 0, 0, 31)
		pg.Visible          = (i==1)
		pg.ClipsDescendants = true
		pages[i] = pg
	end
	local function switchTab(idx)
		for i = 1, #tabNames do
			local col = tabColors and tabColors[i] or C.blue
			pages[i].Visible      = (i==idx)
			tabBtns[i].TextColor3 = (i==idx) and C.txt or C.txtD
			local ul = tabBtns[i]:FindFirstChild("UL")
			if ul then ul.Visible = (i==idx) end
			TweenService:Create(tabBtns[i], TweenInfo.new(0.15),
				{BackgroundColor3=(i==idx) and Color3.fromRGB(18,18,28) or C.tabBg}):Play()
		end
	end
	for i, tb in ipairs(tabBtns) do
		tb.MouseButton1Click:Connect(function() switchTab(i) end)
	end
	return pages, switchTab
end

	local espEnabled = false
	local ESP_CFG    = {username=true, box=true, healthBar=true}
	local espCache   = {}

	local function mkESPLabel(txt, col, sz)
		local l = Instance.new("TextLabel", espSg)
		l.BackgroundTransparency = 1 l.BorderSizePixel = 0
		l.TextColor3 = col l.TextStrokeTransparency = 0.4 l.TextStrokeColor3 = Color3.new(0,0,0)
		l.Font = Enum.Font.GothamBold l.TextScaled = false l.TextSize = sz or 11
		l.ZIndex = 10 l.Text = txt l.Size = UDim2.new(0,120,0,14)
		l.AnchorPoint = Vector2.new(0.5,0.5) l.Visible = false
		return l
	end
	local function mkESPFrame(col, alpha, thick)
		local f = Instance.new("Frame", espSg)
		f.BackgroundColor3 = col f.BackgroundTransparency = alpha or 1
		f.BorderSizePixel = 0 f.ZIndex = 8 f.Visible = false
		if thick then
			local s = Instance.new("UIStroke", f) s.Color = col s.Thickness = thick
		end
		return f
	end
	local function getESPData(plr)
		if espCache[plr] then return espCache[plr] end
		local d = {
			name     = mkESPLabel("", Color3.fromRGB(255,255,255), 11),
			box      = mkESPFrame(Color3.fromRGB(255,60,60), 1, 1.5),
			healthBg = mkESPFrame(Color3.fromRGB(20,20,20), 0, 0),
			healthFg = mkESPFrame(Color3.fromRGB(50,210,80), 0, 0),
		}
		espCache[plr] = d return d
	end
	local function hideESPData(d)
		if not d then return end
		d.name.Visible = false d.box.Visible = false
		d.healthBg.Visible = false d.healthFg.Visible = false
	end
	local function clearESP(plr)
		local d = espCache[plr]
		if not d then return end
		pcall(function() d.name:Destroy() d.box:Destroy() d.healthBg:Destroy() d.healthFg:Destroy() end)
		espCache[plr] = nil
	end

	RunService.RenderStepped:Connect(function()
		if not espEnabled then
			for _, d in pairs(espCache) do hideESPData(d) end
			return
		end
		local cam = workspace.CurrentCamera
		for _, plr in ipairs(Players:GetPlayers()) do
			if plr == player then continue end
			local ch   = plr.Character
			local hrpT = ch and ch:FindFirstChild("HumanoidRootPart")
			local hum  = ch and ch:FindFirstFind("Humanoid")
			if not ch or not hrpT or not hum or hum.Health <= 0 then
				hideESPData(espCache[plr]) continue
			end
			local d = getESPData(plr)
			local sp, onScreen = cam:WorldToViewportPoint(hrpT.Position)
			if not onScreen then hideESPData(d) continue end
			local head = ch:FindFirstChild("Head")
			local hp   = head and cam:WorldToViewportPoint(head.Position + Vector3.new(0,0.5,0)) or sp
			local fp   = cam:WorldToViewportPoint(hrpT.Position - Vector3.new(0,2.5,0))

			local cx    = sp.X
			d.name.Visible = ESP_CFG.username
			if ESP_CFG.username then
				d.name.Text = plr.Name
				d.name.Position = UDim2.new(0, cx, 0, hp.Y - 16)
			end
			d.box.Visible = ESP_CFG.box
			if ESP_CFG.box then
				d.box.Position = UDim2.new(0, cx - charW/2, 0, hp.Y)
				d.box.Size = UDim2.new(0, charW, 0, charH)
				d.box.BackgroundTransparency = 1
